미추홀구 월별 전월세 거래량

rent_data %>% 
  group_by(month = floor_date(계약날짜, "month"), 임대형태) %>% 
  summarise(count=n(), .groups='drop') %>% 
  spread(key='임대형태',value='count') %>% 
  replace(is.na(.), 0) %>% 
  mutate(합계 = 전세 + 월세, 증감률 = (합계 / lag(합계,24) -1 )* 100) %>%
  # View
  plot_ly(x=~month, type="bar", y=~월세, name="월세") %>%
     add_trace(y=~전세, name="전세") %>%
     layout(yaxis = list(title="거래량"), barmode = 'stack') %>% 
  layout(title = '미추홀구 월별 전월세 거래량')

t0 <- rent_data %>% 
  filter("2019-01-01" <= 계약날짜 & 계약날짜 <= "2019-12-31") %>%
  filter(임대형태 == "전세") %>%
  filter(55 <= 전용면적 & 전용면적 <= 60) %>%
  group_by(법정동) %>% 
  summarise(`평균보증금(19년)` = round(mean(보증금액)) ,`거래량(19년)` = n(), .groups='drop')

전월세 월 거래량

rent_data %>%
  filter(아파트 == "인천 SK Sky VIEW") %>%
  group_by(월 =floor_date(계약날짜, "month"), 임대형태) %>%
  summarise(count=n(), .groups='drop') %>%
  spread(key = 임대형태, value = count) %>% 
  replace(is.na(.), 0) %>%
  plot_ly(x=~월, type="bar", y=~월세, name="월세") %>%
    add_trace(y=~전세, name="전세") %>% 
    layout(yaxis = list(title = '거래량'), barmode = 'stack')  %>% 
  layout(title = '인천 SK Sky VIEW 전월세 월 거래량')
