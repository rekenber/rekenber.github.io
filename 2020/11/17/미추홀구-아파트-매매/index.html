미추홀구 34평형 아파트 매매 랭킹

# 아파트 별 전용 면적 당 단가 (전용 면적 84)
data %>% 
  filter(84 < 전용면적 & 전용면적 < 85) %>% 
  group_by(아파트) %>%
  summarise(거래금액평균 = round(mean(거래금액)), 거래횟수 =n(), .groups="drop") %>% 
  arrange(desc(거래금액평균)) %>%
  datatable(options = list(pageLength = 20))

미추홀구 매매가 상위 15위 아파트(34평)

# 상위 15위 아파트 목록 추출
apt_list <- data %>% 
  filter(84 <= 전용면적 & 전용면적 <= 85) %>% 
  group_by(아파트) %>%
  summarise(거래금액평균 = round(mean(거래금액)), 거래횟수 =n(), .groups="drop") %>% 
  arrange(거래금액평균) %>%
  select(아파트) %>%
  tail(15) %>% unlist 

# 아파트 별 전용 면적 당 단가 (전용 면적 84)
data %>% 
  filter(84 < 전용면적 & 전용면적 < 85) %>% 
  filter(아파트 %in% apt_list) %>%
  plot_ly(x=~ 거래금액, color=~아파트, type="box") %>%
  layout(yaxis = list(categoryorder = "array", categoryarray = apt_list))
